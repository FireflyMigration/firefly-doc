```csdiff
  getOrderTotal() {
    let result = 0;
    this.orderDetailsGrid.items.forEach(
      orderDetail =>
        result += orderDetail.quantity.value * orderDetail.unitPrice.value);
    return result.toFixed(2);
  }
+ printCurrentOrder() {
+   window.open(
+     environment.serverUrl + 'home/print/' + this.ordersGrid.currentRow.id.value);
+ }
}
```

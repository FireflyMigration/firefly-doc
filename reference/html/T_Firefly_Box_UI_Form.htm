Tools for dynamically generation UI controls
This example is based on test data. The code for the entities included in this test data can be found in the documentation of Entity
CopyC#
using System;
using System.Collections.Generic;
using System.Drawing;
using System.Text;
using Firefly.Box.Data.Advanced;
using Firefly.Box.UI;

using Firefly.Box;
using Firefly.Box.Testing;
using Button = System.Windows.Forms.Button;

namespace TestFirefly.Box.Documentation
{
    public class UITools
    {
        public void SimpleGrid()
        {
            var jobs = new Pubs.Jobs();
            jobs.InitializeWithTestData();

            var uic = new UIController()
            {
                From = jobs,
                View = UITools.GenerateFormWithGridFor("Display Jobs",
                                                       "A simple grid showing all the rows in the jobs entity",
                                                       jobs.Id,
                                                       jobs.Description,
                                                       jobs.MinLevel,
                                                       jobs.MaxLevel)
            };

            var button = new Button { Text = "Click Me" };
            button.Click += (a, b) => System.Windows.Forms.MessageBox.Show("I Was Clicked");

            UITools.AddControlsToForm(uic.View, button);

            uic.Run();
        }


        public static Grid GenerateGridFor(params ColumnBase[] columns)
        {
            var result = new Grid();
            foreach (var column in columns)
            {
                var gc = new GridColumn
                             {
                                 Text = column.Caption
                             };
                var tb = new TextBox
                             {
                                 Data = column,
                                 Location = new Point(2, 2),
                                 Style = ControlStyle.Flat
                             };
                tb.ResizeToFit(Math.Min(30, column.FormatInfo.MaxLength));
                gc.Controls.Add(tb);
                gc.ResizeToFitContent();
                result.Controls.Add(gc);
            }
            result.ResizeToFitContent(10);
            return result;
        }
        public static Form GenerateFormWithGridFor(string title, string explanation, params ColumnBase[] columns)
        {
            var result = new Form()
                             {
                                 Text = title,
                                 AutoScroll = true
                             };
            var explanationLabel = new Label
                                       {
                                           Location = new Point(5, 5),
                                           Text = explanation,
                                           Multiline = true,
                                           Font = new Font("Arial", 8),
                                           BackColor = SystemColors.Info,
                                           Style = ControlStyle.Standard


                                       };
            explanationLabel.ResizeToFitContent();
            if (explanationLabel.Height > 200)
            {
                var panel = new System.Windows.Forms.Panel
                                {
                                    AutoScroll = true,
                                    Top = 5,
                                    Width = explanationLabel.Width + 20,
                                    Height = 200
                                };
                explanationLabel.Location = new Point(0, 0);
                panel.Controls.Add(explanationLabel);
                AddControlsToForm(result, panel);
            }
            else
            {
                AddControlsToForm(result, explanationLabel);
            }
            if (columns.Length > 0)
                AddControlsToForm(result, GenerateGridFor(columns));
            return result;
        }

        public static void AddControlsToForm(Form form, params System.Windows.Forms.Control[] controls)
        {
            if (form.Controls.Count == 0)
                form.ClientSize = new Size(0, 5);
            int totalWidth = 0;
            foreach (var control in controls)
            {
                totalWidth += control.Width + 5;
            }
            int left = Math.Max(5, form.ClientSize.Width - totalWidth);
            int bottom = 0;

            foreach (var control in controls)
            {
                control.Left = left;
                control.Top = form.ClientSize.Height;
                left = control.Right + 5;
                bottom = Math.Max(bottom, control.Bottom);
                form.Controls.Add(control);
            }
            form.ClientSize = new Size(left, bottom + 5);

        }
    }
